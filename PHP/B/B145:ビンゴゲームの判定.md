<?php
    list($size, $n) = array_map('intval', explode(' ',trim(fgets(STDIN))));
    
    $table = [];
    
    for($i=0;$i<$size;$i++)
    {
        $table[] = array_map('intval', explode(' ',trim(fgets(STDIN))));
    }
    
    $trial = array_map('intval', explode(' ',trim(fgets(STDIN))));
    
    function Judge (array $array, $size)
    {
        $result = 0;
        for($v=0;$v<$size;$v++)
        {
             $new_array = array_unique($array[$v]);
             if(count($new_array) == 1 && $new_array[0] == 'X')
             {
                 $result += 1;
             }
        }
        
        return $result;
    }
    
    //置換
    foreach ($trial as $tr)
    {
        for($j=0;$j<$size;$j++)
        {
            for($k=0;$k<$size;$k++)
            {
                if($table[$j][$k] === $tr || $table[$j][$k] === 0)
                {
                    $table[$j][$k] = 'X';
                }
            }
        }
    }
    
    //元の配列を縦で見た時の配列に書き換え
    $vertical = [];
    $small_vertical = [];
    for($x=0;$x<$size;$x++)
    {
        $small_vertical = [];
        for($y=0;$y<$size;$y++)
        {
            $small_vertical[] = $table[$y][$x];
        }
        $vertical[] = $small_vertical;
    }
    
    //斜め方向の配列取得([1,1],[2,2]...)
    $diagonal1 = [];
    for($m=0;$m<$size;$m++)
    {
        $diagonal1[] = $table[$m][$m];
    }
    
    //斜め方向の配列取得([1,3],[2,2]...)
    $diagonal2 = [];
    for($n=0;$n<$size;$n++)
    {
        for($l=0;$l<$size;$l++)
        {
            if(($n+1)+($l+1) == $size+1)
            {
                //var_dump($n+$l);
                $diagonal2[] = $table[$n][$l];
            }
        }
    }

    
    print_r(Judge($vertical, $size) + Judge([$diagonal1], 1) + Judge([$diagonal2], 1) + Judge($table, $size));
?>
