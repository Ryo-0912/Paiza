<?php
$def = (string)trim(fgets(STDIN));

function isPalindrome($str)
{
    return $str === strrev($str);
}


while (!isPalindrome($def)) {
    $reverse_def = strrev($def);
    $new_num = (int)$def + (int)$reverse_def;
    $def = (string)$new_num;

    if (isPalindrome($def)) {
