<?php
    list($m , $n, $l) = array_map('intval', explode(' ', trim(fgets(STDIN))));
    
    $list = [];
    for($i=0;$i<$m;$i++)
    {
        $list[] = trim(fgets(STDIN));
    }
    
    if($m >= $n)
    {
        sort($list);
    
        $list = array_splice($list, $l);
          
        print_r(array_sum($list));
        
    } else {
        print_r(array_sum($list));
    }
?>
