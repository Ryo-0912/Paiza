<?php
    $n = (int)trim(fgets(STDIN));
    
    $info = [];
    
    for($i=0;$i<$n;$i++)
    {
        $info[] =  explode(' ', trim(fgets(STDIN)));
    }
    
    $result = 0;
    foreach ($info as $i)
    {
        if($i[0] == 's')
        {
            $score = array_diff($i, [$i[0]]);
            if(array_sum($score) >= 350 && $score[2] + $score[3] >= 160){
                $result++;
            }
        } else {
            $score = array_diff($i, [$i[0]]);
            if(array_sum($score) >= 350 && $score[4] + $score[5] >= 160){
                $result++;
            }
        }
    }
    
    print_r($result);  
?>
