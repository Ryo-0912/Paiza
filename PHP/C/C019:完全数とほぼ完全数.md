<?php
    $n = trim(fgets(STDIN));
    
    $target = array();
    
    for($i=0;$i<$n;$i++)
    {
        array_push($target, trim(fgets(STDIN)));
    }
    
    $target = array_map('intval', $target);
    
    $a = array();
    foreach($target as $t)
    {
        // ここで$aを初期化しないと、前回の数字の約数を引き継いでしまう
        $a = array();
        for($j=1;$j<$t;$j++)
        {
            if($t%$j === 0)
            {
                array_push($a, $j);
            }
        }

        $sum = array_sum($a);
        
        if($sum === $t)
        {
            echo "perfect\n" ;
        } elseif (abs($sum - $t) === 1)
        {
            echo "nearly\n";
        } else {
            echo "neither\n";
        }
    }
?>
