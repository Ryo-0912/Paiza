    $target_number = trim(fgets(STDIN));
    
    $n = (int)trim(fgets(STDIN));
    
    $a = [];
    for($i=0;$i<$n;$i++)
    {
        $room_number = trim(fgets(STDIN));
        if(!str_contains($room_number, $target_number))
        {
            array_push($a, $room_number);
        } else {
            array_push($a, "none");
        }
    }
    
    if(empty(array_diff($a, ["none"])))
    {
        print_r("none");
    } elseif (in_array("none", $a)) {
        foreach ($a as $x)
        {
            if($x !== "none")
            {
                print_r($x . "\n");
            }
        }
    } 
?>
