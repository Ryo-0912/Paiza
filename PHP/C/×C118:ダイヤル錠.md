    list($d, $n, $l) = array_map('intval', explode(' ', trim(fgets(STDIN))));

        if($sum == 0 && $position == 0)
        {
            $sum += $li;

            $a = abs($li - $position);
            $b = $d - abs($li - $position);
            if($a - $b > 0)
            {
                $sum += $b;
            } else {
                $sum += $a;
            }
        }
        $position = $li;
    }

    if($sum >= $l)

        echo("No\n");
