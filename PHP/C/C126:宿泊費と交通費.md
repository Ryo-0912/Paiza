    list($sinkansen, $hotel, $n) = array_map('intval', explode(' ', trim(fgets(STDIN))));
    
    
    $date = [];
    $total = 0;
    
    for($i=0;$i<$n;$i++)
    {
        $date[] = array_map('intval', explode(' ', trim(fgets(STDIN))));
    }
    
    for($j=0;$j<$n-1;$j++)
    {
        $term =  $date[$j+1][0] - $date[$j][1];
        if($sinkansen *  2 >= $hotel * $term)
        {
            $total += $hotel * $term;
        } else {
            $total += $sinkansen * 2;
        }
    }
    
    $result = 2 * $sinkansen + $total;
    
    print_r($result);
?>
