<?php
    $n = (int)trim(fgets(STDIN));

    $jyannken = ['rock' , 'scissors' , 'paper'];
    for($i=0;$i<$n;$i++)
    {
        $x = trim(fgets(STDIN));
        $jyannken = array_diff($jyannken, [$x]);
    }
    
    if(count($jyannken) == 0 || count($jyannken) == 2)
    {
        echo('draw');
    } elseif(count($jyannken) == 1 && $jyannken[0] == 'rock') {
        echo('scissors');
    } elseif(count($jyannken) == 1 && $jyannken[1] == 'scissors') {
        echo('paper');
    } elseif(count($jyannken) == 1 && $jyannken[2] == 'paper') {
        echo('rock');
    }
?>

↓これでもいい
```
<?php
    $n = (int)trim(fgets(STDIN));

    $jyannken = ['rock', 'scissors', 'paper'];
    for ($i = 0; $i < $n; $i++) {
        $x = trim(fgets(STDIN));
        $jyannken = array_diff($jyannken, [$x]);
    }
    
    $jyannken = array_values($jyannken); // 配列のキーを再割り当て

    if (count($jyannken) == 0 || count($jyannken) == 2) {
        echo 'draw';
    } elseif ($jyannken[0] == 'rock') {
        echo 'scissors';
    } elseif ($jyannken[0] == 'scissors') {
        echo 'paper';
    } elseif ($jyannken[0] == 'paper') {
        echo 'rock';
    }
?>
```
