[PHP_INT_MINとPHP_INT_MAX](https://github.com/Ryo-0912/laravel-learning/blob/main/PHP_INT_MIN%E3%81%A8PHP_INT_MAX.md)
```
<?php
    $n = (int)trim(fgets(STDIN));
    
    $values = [];
    $max = PHP_INT_MIN;
    $min = PHP_INT_MAX;
    
    for ($i = 0; $i < $n; $i++) {
        $values[] = $line = array_map('intval', explode(' ', trim(fgets(STDIN))));
        $max = max($max, $line[2]);　　　　　　// 第一引数の値と第二引数の値を比較して、大きい方を出力
        $min = min($min, $line[3]);   // 第一引数の値と第二引数の値を比較して、小さい方を出力
    }
    
    $start = $values[0][0];
    $end = $values[$n - 1][1];
    
    echo "$start $end $max $min";
?>
```

↓変数などに入れず、一括で出力まで行う
```
<?php
    $n = (int)trim(fgets(STDIN));
    
    $values = [];
    $max = PHP_INT_MIN;
    $min = PHP_INT_MAX;
    
    for ($i = 0; $i < $n; $i++) {
        $line = array_map('intval', explode(' ', trim(fgets(STDIN))));
        $values[] = $line;
        $max = max($max, $line[2]);
        $min = min($min, $line[3]);
    }
    
    // $start, $end を直接出力時に取得
    echo "{$values[0][0]} {$values[$n - 1][1]} $max $min";
?>
