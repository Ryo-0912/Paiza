<?php
    $n = (int)trim(fgets(STDIN));
    
    $food = [];
    $book = [];
    $clothes = [];
    $other = [];
    
    for($i=0;$i<$n;$i++)
    {
        list($item_num, $price) = explode(' ', trim(fgets(STDIN)));
        if($item_num == 0)
        {
            $food[] = $price;
        } elseif ($item_num == 1) {
            $book[] = $price;
        } elseif ($item_num == 2) {
            $clothes[] = $price;
        } else {
            $other[] = $price;
        }
    }
    
    $point = floor((array_sum($food)/100))*5 + floor((array_sum($book)/100))*3 + floor((array_sum($clothes)/100))*2 + floor(array_sum($other)/100);
    
    print_r($point);
?>
