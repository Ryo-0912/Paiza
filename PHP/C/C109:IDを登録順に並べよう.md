<?php
    $n = (int)trim(fgets(STDIN));
    
    $target = [];
    for($i=0;$i<$n;$i++)
    {
        array_push($target, trim(fgets(STDIN)));
    }
    
    $array = [];
    foreach($target as $t)
    {
        $length = mb_strlen($t);
        $s = "";
        
        for($j=0;$j<$length;$j++)
        {
            if(ctype_digit($t[$j]))
            {
                $s .= $t[$j];
            }
        }
        
        array_push($array, $s);
    }
    
    $sx = array_values($array);

    
    asort($sx);
    
    $new_sx = array_keys($sx);
    
    $result = [];
    foreach($new_sx as $ns)
    {
        array_push($result, $target[$ns]);
    }
    
    foreach($result as $r)
    {
        print_r($r . "\n");
    }
    
?>
